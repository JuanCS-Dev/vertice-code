Integration Examples

Connect your apps in Python, JavaScript, or cURL

import requests

# MCP Server endpoint
MCP_URL = "https://vertice-mcp-server-452089804714.us-central1.run.app/mcp"

# List available tools
response = requests.post(MCP_URL, json={
    "jsonrpc": "2.0",
    "method": "tools/list",
    "id": "python-client-1"
})

tools = response.json()
print(f"Available tools: {len(tools['result'])}")

# Call a specific tool
response = requests.post(MCP_URL, json={
    "jsonrpc": "2.0",
    "method": "tools/call",
    "params": {
        "name": "read_file",
        "arguments": {"path": "/path/to/file.txt"}
    },
    "id": "python-client-2"
})

result = response.json()
print(result['result'])
// MCP Server endpoint
const MCP_URL = "https://vertice-mcp-server-452089804714.us-central1.run.app/mcp";

// List available tools
const listTools = async () => {
    const response = await fetch(MCP_URL, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            jsonrpc: '2.0',
            method: 'tools/list',
            id: 'js-client-1'
        })
    });

    const data = await response.json();
    console.log(`Available tools: ${data.result.length}`);
    return data.result;
};

// Call a specific tool
const callTool = async (toolName, args) => {
    const response = await fetch(MCP_URL, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            jsonrpc: '2.0',
            method: 'tools/call',
            params: { name: toolName, arguments: args },
            id: 'js-client-2'
        })
    });

    const result = await response.json();
    return result.result;
};

    "jsonrpc": "2.0",
    "method": "tools/list",

  }'
